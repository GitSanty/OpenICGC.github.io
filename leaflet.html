    height: 500px;
    width: 100%;

  <script>
  function init() {
    var map = L.map('map', {
      center: [41.6863, 2.4382],
      zoom: 10,
      attribution: 'Institut Cartogràfic i Geològic de Catalunya CC-BY-SA-3'
    });
    var topoICGC = L.tileLayer('https://geoserveis.icgc.cat/icc_mapesmultibase/noutm/wmts/topo/GRID3857/{z}/{x}/{y}.jpeg', {
      maxZoom: 19
    });

    var topoICGCWEB = L.tileLayer('https://tilemaps.icgc.cat/mapfactory/wmts/topo_suau/CAT3857/{z}/{x}/{y}.png', {
      maxZoom: 18
    }).addTo(map);

    var topoICGCDark = L.tileLayer('https://tilemaps.icgc.cat/mapfactory/wmts/nit_topo_suau/CAT3857/{z}/{x}/{y}.png', {
      maxZoom: 18
    });

    var ortoICGC =
      L.tileLayer('https://geoserveis.icgc.cat/icc_mapesmultibase/noutm/wmts/orto/GRID3857/{z}/{x}/{y}.jpeg', {
        maxZoom: 19
      });
    var hibridICGC = L.tileLayer('https://tilemaps.icgc.cat/mapfactory/wmts/hibrida_total/CAT3857/{z}/{x}/{y}.png', {
      maxZoom: 18
    });
    var hibrida = L.layerGroup([ortoICGC, hibridICGC]);
    var mapaBase = {
      'Clàssic': topoICGC,
      'Web': topoICGCWEB,
      'Fosc': topoICGCDark,
      'Híbrid': hibrida
    };
    controlCapas = L.control.layers(mapaBase, null, {
      collapsed: false
    });
    controlCapas.addTo(map)
  }
  </script>
 </head>
 <body onLoad="init()">
  <div id="map">  </div>
